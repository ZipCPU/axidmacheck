@PREFIX=controlbus
@BUS.NAME=axil
@BUS.TYPE=axil
@BUS.CLOCK=clk
@BUS.RESET=!i_reset
@BUS.WIDTH=32
@MASTER.BUS=axil
@MASTER.TYPE=SUBBUS
@SLAVE.BUS=axi
@SLAVE.TYPE=OTHER
@MAIN.INSERT=
	axi2axilite #(
		.C_AXI_ADDR_WIDTH(@$MASTER.BUS.AWID),
		.C_AXI_DATA_WIDTH(@$MASTER.BUS.WIDTH),
		.C_AXI_ID_WIDTH(@$(SLAVE.BUS.IDWIDTH))
	) @$(PREFIX)i (
		.S_AXI_ACLK(@$(SLAVE.BUS.CLOCK.WIRE)),
		.S_AXI_ARESETN(@$(SLAVE.BUS.RESET)),
		//
		@$(SLAVE.ANSIPORTLIST),
		//
		@$(MASTER.ANSIPORTLIST)
	);
